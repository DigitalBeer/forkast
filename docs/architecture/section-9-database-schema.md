# Section 9: Database Schema

```sql
CREATE TYPE public.subscription_status AS ENUM ('free', 'premium');

CREATE TABLE public.profiles (
  id uuid NOT NULL PRIMARY KEY REFERENCES auth.users(id) ON DELETE CASCADE,
  full_name text,
  subscription_status public.subscription_status DEFAULT 'free' NOT NULL,
  stripe_customer_id text,
  subscription_current_period_end timestamptz,
  created_at timestamptz DEFAULT now()
);

CREATE TABLE public.meals (
  id bigint GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  user_id uuid NOT NULL REFERENCES public.profiles(id) ON DELETE CASCADE,
  name text NOT NULL,
  meal_type text NOT NULL,
  image_url text,
  source_url text,
  ingredients text,
  instructions text,
  rating smallint CHECK (rating >= 1 AND rating <= 5),
  dietary_tags text[],
  created_at timestamptz DEFAULT now()
);
CREATE INDEX ON public.meals (user_id);

CREATE TABLE public.meal_plans (
  id bigint GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  user_id uuid NOT NULL REFERENCES public.profiles(id) ON DELETE CASCADE,
  start_date date NOT NULL,
  end_date date NOT NULL,
  created_at timestamptz DEFAULT now()
);
CREATE INDEX ON public.meal_plans (user_id);

CREATE TABLE public.planned_meals (
  id bigint GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  meal_plan_id bigint NOT NULL REFERENCES public.meal_plans(id) ON DELETE CASCADE,
  meal_id bigint NOT NULL REFERENCES public.meals(id) ON DELETE CASCADE,
  planned_for_date date NOT NULL,
  action_type text NOT NULL,
  created_at timestamptz DEFAULT now()
);
CREATE INDEX ON public.planned_meals (meal_plan_id);
CREATE INDEX ON public.planned_meals (meal_id);
```
